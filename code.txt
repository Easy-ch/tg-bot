#include <iostream>
#include <iomanip>
#include <cstdlib>
#include <ctime>    
using namespace std;

int main() {
    setlocale(0, "rus");
    srand(time(NULL));
    int M[5][5];

    cout << "Задать двумерный числовой массив произвольной размерности. Найти сумму\nположительных элементов каждого столбца массива." << endl;
    cout << endl << "Исходный массив:" << endl;

    for (int i = 0; i < 5; i++)
    {
        for (int j = 0; j < 5; j++)
        {
            M[i][j] = -12 + rand() % 28; // заполнение массива значениями в диапазоне [-12, 28]
            cout << setw(4) << M[i][j];
        }
        cout << endl;
    }

    cout << endl;

    for (int i = 0; i < 5; i++)
    {
        int S = 0;
        for (int j = 0; j < 5; j++)
        {
            if (M[j][i] > 0)
                S += M[j][i];
        }
        cout << "Сумма положительных элементов в " << i+1 << " столбце массива:" << S << endl;
    }
    
    system("pause>>void");
    return 0;
}
